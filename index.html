              void push(Stack *stack, int value) {
                if (isFull(stack)) {
                    printf("Error: La pila está llena. No se puede realizar el push.\n");
                    return;
                }
            
                stack->top++;
                stack->items[stack->top] = value;
            }
            
        
            
              int pop(Stack *stack) {
                if (isEmpty(stack)) {
                    printf("Error: La pila está vacía. No se puede realizar el pop.\n");
                    exit(EXIT_FAILURE);
                }
            
                int poppedValue = stack->items[stack->top];
                stack->top--;
            
                return poppedValue;
            }
